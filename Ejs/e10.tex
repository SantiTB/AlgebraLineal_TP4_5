\item Sea $p(x)=1+x+x^2+x^3$.
    \begin{enumerate}
        \item Calcular $p(A)$ para \[A=\begin{pmatrix}
                1&-1\\-1&2
            \end{pmatrix}.\]
            \begin{mdframed}[style=s]
                \begin{center}
                    $p(A)=I+A+A^2+A^3=\begin{pmatrix}
                        9&-12\\-12&21
                    \end{pmatrix}$
                \end{center}
            \end{mdframed}
        \item Probar que $A$ y $p(A)$ conmutan.
            \begin{mdframed}[style=s]
                \begin{center}
                    $Ap(A)=p(A)A=\begin{pmatrix}
                        21&-33\\-33&54
                    \end{pmatrix}$
                \end{center}
            \end{mdframed}
    \end{enumerate}
